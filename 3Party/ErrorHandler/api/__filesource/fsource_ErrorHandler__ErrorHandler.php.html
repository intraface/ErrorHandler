  49.     public function addObserver($observer, $errorlevel = E_USER_NOTICE)
  50.     {
  51.         if (!method_exists($observer, 'update')) {
  52.             throw new Exception('The observer ' . get_class($logger) . ' does not implement an updated method.');

  62.     protected function getObservers()

  65.     }
  66.  
  67.     /**

  74.     protected function notifyObservers($error)

  80.         }
  81.     }

  95.         // Error types
  96.         $error_types = array(
  97.             1 => 'ERROR',
  98.             2 => 'WARNING',
  99.             4 => 'PARSE',
  100.             8 => 'NOTICE',
  101.             16 => 'CORE_ERROR',
  102.             32 => 'CORE_WARNING',
  103.             64 => 'COMPILE_ERROR',

  105.             256 => 'USER_ERROR',
  106.             512 => 'USER_WARNING',
  107.             1024 => 'USER_NOTICE',
  108.             2047 => 'ALL',
  109.             2048 => 'STRICT',
  110.             4096 => 'RECOVERABLE_ERROR'
  111.         );
  112.  
  113.         // Filling up details
  114.         $details['date']    = date('Y-m-d H:i:s');

  116.         $details['code']    = $error_types[$errno];
  117.         $details['errno']   = $errno;
  118.         $details['message'] = preg_replace("%\s\[<a href='function\.[\d\w-_]+'>function\.[\d\w-_]+</a>\]%", '', $errstr); // Removing PHP function links
  119.         $details['line']    = $errline;
  120.         $details['file']    = $errfile;

  123.

  126.         for($i = 1; $i < sizeof($backtrace); $i++) {
  127.             $details['trace'][$i - 1]['file']     = @$backtrace[$i]['file'];
  128.             $details['trace'][$i - 1]['line']     = @$backtrace[$i]['line'];

  130.             $details['trace'][$i - 1]['class']    = @$backtrace[$i]['class'];
  131.             $details['trace'][$i - 1]['type']     = @$backtrace[$i]['type'];
  132.             $details['trace'][$i - 1]['args']     = @$backtrace[$i]['args'];
  133.         }
  134.         return self::handleAny($details);
